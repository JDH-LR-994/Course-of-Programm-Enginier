700template<typename T, typename T0>
701bool RedBlackTree<T, T0>::remove(const T &key) {
702 NodePtr node = searchNode(key);
703 if (node == NIL) return false; // Узел не найден
704
705 NodePtr successor = NIL;
706 NodePtr child = NIL;
707 Color originalColor = node->color_;
708
709 // Определяем узел для удаления и его потомка
710 if (node->left_ == NIL) {
711 child = node->right_;
712 replaceSubtree(node, child);
713 } else if (node->right_ == NIL) {
714 child = node->left_;
715 replaceSubtree(node, child);
716 } else {
717 // У узла есть оба потомка
718 successor = minimum(node->right_);
719 originalColor = successor->color_;
720 child = successor->right_;
721
722 if (successor->parent_.lock() != node) {
723 replaceSubtree(successor, child);
724 successor->right_ = node->right_;
725 successor->right_->parent_ = successor;
726 }
727 replaceSubtree(node, successor);
728 successor->left_ = node->left_;
729 successor->left_->parent_ = successor;
730 successor->color_ = node->color_;
731 }
732
733 // Балансировка, если удалён чёрный узел
734 if (originalColor == Color::BLACK) {
735 fixDelete(child);
736 }
737 countNodes--;

860template<typename T, typename T0>
861void RedBlackTree<T, T0>::fixInsert(NodePtr pivot) {
862 while (pivot != root && pivot->parent_.lock()->color_ == Color::RED) {
863 auto parent = pivot->parent_.lock();
864 auto grandparent = parent->parent_.lock();
865 NodePtr uncle = (parent == grandparent->left_) ? grandparent->right_ : grandparent->left_;
866
867 if (uncle->color_ == Color::RED) {
868 // Случай 1: дядя красный.
869 parent->color_ = Color::BLACK;
870 uncle->color_ = Color::BLACK;
871 grandparent->color_ = Color::RED;
872 pivot = grandparent;
873 } else {
874 if (parent == grandparent->left_) {
875 // Случай 2: pivot — правый потомок.
876 if (pivot == parent->right_) {
877 pivot = parent;
878 rotateLeft(pivot);
879 parent = pivot->parent_.lock(); // Обновляем parent после поворота
880 }
881 // Случай 3: pivot — левый потомок.
882 parent->color_ = Color::BLACK;
883 grandparent->color_ = Color::RED;
884 rotateRight(grandparent);
885 } else {
886 // Зеркальные случаи для правого поддерева.
887 if (pivot == parent->left_) {
888 pivot = parent;
889 rotateRight(pivot);
890 parent = pivot->parent_.lock(); // Обновляем parent
891 }
892 parent->color_ = Color::BLACK;
893 grandparent->color_ = Color::RED;
894 rotateLeft(grandparent);
895 }
896 }
897 }
898 root->color_ = Color::BLACK;

943template<typename T, typename T0>
944void RedBlackTree<T, T0>::fixDelete(NodePtr pivot) {
945 while (pivot != root && pivot->color_ == Color::BLACK) {
946 auto parent = pivot->parent_.lock();
947 if (!parent) break; // Защита от nullptr
948
949 bool isLeftChild = (pivot == parent->left_);
950 NodePtr sibling = isLeftChild ? parent->right_ : parent->left_;
951
952 // Случай 1: Брат красный -> преобразуем в случаи 2-4
953 if (sibling != NIL && sibling->color_ == Color::RED) {
954 sibling->color_ = Color::BLACK;
955 parent->color_ = Color::RED;
956 if (isLeftChild) {
957 rotateLeft(parent);
958 sibling = parent->right_; // Обновляем брата после поворота
959 } else {
960 rotateRight(parent);
961 sibling = parent->left_;
962 }
963 }
964
965 // Случай 2: Брат чёрный, оба потомка чёрные
966 if ((sibling->left_ == NIL || sibling->left_->color_ == Color::BLACK) &&
967 (sibling->right_ == NIL || sibling->right_->color_ == Color::BLACK)) {
968 if (sibling != NIL) {
969 sibling->color_ = Color::RED;
970 }
971 pivot = parent; // Поднимаем проблему вверх
972 } else {
973 // Случай 3: Брат чёрный, ближний потомок красный
974 if (isLeftChild && (sibling->right_ == NIL || sibling->right_->color_ == Color::BLACK)) {
975 if (sibling->left_ != NIL) {
976 sibling->left_->color_ = Color::BLACK;
977 }
978 sibling->color_ = Color::RED;
